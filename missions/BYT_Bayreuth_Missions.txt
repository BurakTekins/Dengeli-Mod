# Dengeli - Bayreuth (BYT) combined Franconian and custom mission layout
#
# The shared Franconian mission series use slots 2-4. BYT is excluded from
# those shared series and receives the equivalent missions here, together with
# its own lower tree. Keeping one active series per slot prevents EU4 from
# discarding missions or collapsing every custom mission into one column.
#
# Slots:       2                         3                         4
# Row 1                                  Build an Army
# Row 2        Conquer Nuremberg         Conquer Wurzburg         Conquer Ansbach
# Row 3                                  Unifying Franconia
# Row 4        Strength in Unity                                   Growing Cities
# Row 5        Emperor's Recognition                               Raubritter
#
# Row 7                                  Iron Heart
# Row 8        Developed Realm           Workshops                 Pious Principality
# Row 9                                  Centres of Commerce
# Row 10                                 Stone Bridge Network

byt_combined_missions_2 = {
	slot = 2
	generic = no
	ai = yes
	potential = {
		tag = BYT
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes

	byt_fkn_conquer_nuremburg = {
		icon = sources_of_trade
		required_missions = { byt_fkn_build_an_army }
		position = 2
		provinces_to_highlight = {
			OR = {
				province_id = 67
				province_id = 4715
			}
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		}
		trigger = {
			OR = {
				67 = { country_or_non_sovereign_subject_holds = ROOT }
				4715 = { country_or_non_sovereign_subject_holds = ROOT }
			}
		}
		effect = {
			add_country_modifier = {
				name = FKN_conquest_of_nuremburg
				duration = 10950
			}
		}
	}

	byt_fkn_strength_in_unity = {
		icon = catholic_confraternity
		required_missions = { }
		position = 4
		trigger = {
			calc_true_if = {
				all_ally = {
					has_opinion = {
						who = ROOT
						value = 150
					}
				}
				amount = 2
			}
		}
		effect = {
			add_dip_power = 50
			custom_tooltip = PRU_gain_10_mutual_trust
			hidden_effect = {
				every_ally = {
					add_trust = {
						who = ROOT
						value = 10
						mutual = yes
					}
				}
			}
		}
	}

	byt_fkn_recognition_of_emperor = {
		icon = switzerland_and_the_empire
		required_missions = { byt_fkn_strength_in_unity }
		position = 5
		provinces_to_highlight = {
			owned_by = emperor
		}
		trigger = {
			OR = {
				emperor = {
					has_opinion = {
						who = ROOT
						value = 125
					}
				}
				is_emperor = yes
				NOT = { hre_size = 1 }
			}
		}
		effect = {
			add_country_modifier = {
				name = FKN_imperial_diplomacy
				duration = 7300
			}
			custom_tooltip = FKN_emperor_tt
			tooltip = {
				add_country_modifier = {
					name = FKN_emperor_recogntion
					duration = 10950
				}
			}
			hidden_effect = {
				if = {
					limit = {
						NOT = { has_global_flag = FKN_emperor_mission_taken }
					}
					add_country_modifier = {
						name = FKN_emperor_recogntion
						duration = 10950
					}
					set_global_flag = FKN_emperor_mission_taken
				}
			}
		}
	}

	byt_developed_realm = {
		icon = mission_modernize_production
		position = 8
		required_missions = { byt_iron_heart }
		provinces_to_highlight = {
			owned_by = ROOT
			NOT = { base_production = 10 }
		}
		trigger = {
			calc_true_if = {
				all_owned_province = { base_production = 10 }
				amount = 10
			}
		}
		effect = {
			custom_tooltip = byt_developed_realm_dev_tt
			hidden_effect = {
				random_owned_province = { limit = { is_core = ROOT } add_base_production = 1 }
				random_owned_province = { limit = { is_core = ROOT } add_base_production = 1 }
				random_owned_province = { limit = { is_core = ROOT } add_base_production = 1 }
				random_owned_province = { limit = { is_core = ROOT } add_base_production = 1 }
				random_owned_province = { limit = { is_core = ROOT } add_base_production = 1 }
				random_owned_province = { limit = { is_core = ROOT } add_base_production = 1 }
				random_owned_province = { limit = { is_core = ROOT } add_base_production = 1 }
				random_owned_province = { limit = { is_core = ROOT } add_base_production = 1 }
				random_owned_province = { limit = { is_core = ROOT } add_base_production = 1 }
				random_owned_province = { limit = { is_core = ROOT } add_base_production = 1 }
			}
			add_country_modifier = {
				name = byt_industrial_boom
				duration = 3650
			}
		}
	}
}

					NOT = { owned_by = ROOT }

	byt_fkn_unifying_franconia = {
		icon = influence_the_princes
		required_missions = {
			byt_fkn_conquer_wurzburg
			byt_fkn_conquer_nuremburg
			byt_fkn_conquer_ansbach
		}
		position = 3
		provinces_to_highlight = {
			OR = {
				area = franconia_area
				area = upper_franconia_area
			}
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		}
		trigger = {
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				OR = {
					area = franconia_area
					area = upper_franconia_area
				}
				value = 5
			}
		}
		effect = {
			custom_tooltip = FKN_every_franconian_province
			hidden_effect = {
				every_owned_province = {
					limit = { culture = franconian }
					add_nationalism = -10
				}
			}
			random_owned_province = {
				limit = { area = franconia_area }
				add_prosperity = 50
			}
			random_owned_province = {
				limit = { area = upper_franconia_area }
				add_prosperity = 50
			}
			add_yearly_manpower = 1
		}
	}

	byt_iron_heart = {

		position = 7
		required_missions = { }

			2965 = { add_province_modifier = { name = byt_iron_forges duration = 5475 } }

	}

	byt_workshops_of_franconia = {
		icon = mission_have_manufactories

		provinces_to_highlight = {
			owned_by = ROOT

		}
		trigger = {

				all_owned_province = { has_building = workshop }

				name = byt_efficient_construction

	byt_trade_centers = {
		icon = mission_dominate_home_trade_node
		position = 9
		required_missions = {
			byt_developed_realm
			byt_workshops_of_franconia
			byt_pious_principality
		}
		provinces_to_highlight = {
			owned_by = ROOT
			center_of_trade = 3
			NOT = { has_building = trade_depot }
		}
		trigger = {
			calc_true_if = {
				all_owned_province = { center_of_trade = 3 }
				amount = 3
			}
		}
		effect = {
			every_owned_province = {
				limit = {
					center_of_trade = 3
					NOT = { has_building = trade_depot }
				}
				add_building = trade_depot
			}
		}
	}

	byt_stone_bridge_network = {
		icon = mission_develop_the_capital
		position = 10
		required_missions = { byt_trade_centers }
		provinces_to_highlight = {
			province_id = 2957
			NOT = { has_great_project = { tier = 3 type = gme_regensburg_stone_bridge } }
		}
		trigger = {
			2957 = {
				country_or_non_sovereign_subject_holds = ROOT
				has_great_project = { tier = 3 type = gme_regensburg_stone_bridge }
			}
		}
		effect = {
			every_province = {
				limit = {
					OR = {
						area = lower_bavaria_area
						area = upper_franconia_area
					}
				}
				add_province_modifier = {
					name = byt_regensburg_bridge_network
					duration = -1
				}
			}
		}
	}
